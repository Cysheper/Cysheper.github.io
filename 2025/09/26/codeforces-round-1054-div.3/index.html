void solve() {
    int n; cin >> n;
    string s; cin >> s;
    int cnta = count(all(s), 'a');
    int idx = (cnta + 1) / 2;
    int mid = 0;
    int cnt = 0;
    for(int i = 0; i < n; ++ i) {
        if(s[i] == 'a') {
            cnt ++;
        }
        if(cnt == idx) {
            mid = i;
            break;
        }
    }
    int l = mid, r = mid;
    int ans1 = 0;
    for(int i = mid; i >= 0; -- i) {
        if(s[i] == 'a') {
            ans1 += abs(l - i);
            l --;
        }
    }
    for(int i = mid; i < n; ++ i) {
        if(s[i] == 'a') {
            ans1 += abs(i - r);
            r ++;
        }
    }
    int cntb = count(all(s), 'b');
    idx = (cntb + 1) / 2;
    mid = 0;
    cnt = 0;
    for(int i = 0; i < n; ++ i) {
        if(s[i] == 'b') {
            cnt ++;
        }
        if(cnt == idx) {
            mid = i;
            break;
        }
    }
    int ans2 = 0;
    l = mid, r = mid;
    for(int i = mid; i >= 0; -- i) {
        if(s[i] == 'b') {
            ans2 += abs(l - i);
            l --;
        }
    }
    for(int i = mid; i < n; ++ i) {
        if(s[i] == 'b') {
            ans2 += abs(i - r);
            r ++;
        }
    }
    cout << min(ans1, ans2) << "\n";
}
